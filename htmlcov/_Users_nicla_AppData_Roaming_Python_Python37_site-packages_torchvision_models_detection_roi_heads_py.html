18def fastrcnn_loss(class_logits, box_regression, labels, regression_targets): 

19 # type: (Tensor, Tensor, List[Tensor], List[Tensor]) 

20 """ 

21 Computes the loss for Faster R-CNN. 

22 

23 Arguments: 

24 class_logits (Tensor) 

25 box_regression (Tensor) 

26 labels (list[BoxList]) 

27 regression_targets (Tensor) 

28 

29 Returns: 

30 classification_loss (Tensor) 

31 box_loss (Tensor) 

32 """ 

33 

34 labels = torch.cat(labels, dim=0) 

35 regression_targets = torch.cat(regression_targets, dim=0) 

36 

37 classification_loss = F.cross_entropy(class_logits, labels) 

38 

39 # get indices that correspond to the regression targets for 

40 # the corresponding ground truth labels, to be used with 

41 # advanced indexing 

42 sampled_pos_inds_subset = torch.nonzero(labels > 0).squeeze(1) 

43 labels_pos = labels[sampled_pos_inds_subset] 

44 N, num_classes = class_logits.shape 

45 box_regression = box_regression.reshape(N, -1, 4) 

46 

47 box_loss = F.smooth_l1_loss( 

48 box_regression[sampled_pos_inds_subset, labels_pos], 

49 regression_targets[sampled_pos_inds_subset], 

50 reduction="sum", 

51 ) 

52 box_loss = box_loss / labels.numel() 

53 

54 return classification_loss, box_loss

57def maskrcnn_inference(x, labels): 

58 # type: (Tensor, List[Tensor]) 

59 """ 

60 From the results of the CNN, post process the masks 

61 by taking the mask corresponding to the class with max 

62 probability (which are of fixed size and directly output 

63 by the CNN) and return the masks in the mask field of the BoxList. 

64 

65 Arguments: 

66 x (Tensor): the mask logits 

67 labels (list[BoxList]): bounding boxes that are used as 

68 reference, one for ech image 

69 

70 Returns: 

71 results (list[BoxList]): one BoxList for each image, containing 

72 the extra field mask 

73 """ 

74 mask_prob = x.sigmoid() 

75 

76 # select masks coresponding to the predicted classes 

77 num_masks = x.shape[0] 

78 boxes_per_image = [len(l) for l in labels] 

79 labels = torch.cat(labels)

141def keypoints_to_heatmap(keypoints, rois, heatmap_size): 

142 # type: (Tensor, Tensor, int) 

143 offset_x = rois[:, 0] 

144 offset_y = rois[:, 1] 

145 scale_x = heatmap_size / (rois[:, 2] - rois[:, 0]) 

146 scale_y = heatmap_size / (rois[:, 3] - rois[:, 1]) 

147 

148 offset_x = offset_x[:, None] 

149 offset_y = offset_y[:, None] 

150 scale_x = scale_x[:, None] 

151 scale_y = scale_y[:, None] 

152 

153 x = keypoints[..., 0] 

154 y = keypoints[..., 1] 

155 

156 x_boundary_inds = x == rois[:, 2][:, None] 

157 y_boundary_inds = y == rois[:, 3][:, None] 

158 

159 x = (x - offset_x) * scale_x 

160 x = x.floor().long() 

161 y = (y - offset_y) * scale_y 

162 y = y.floor().long() 

163 

164 x[x_boundary_inds] = torch.tensor(heatmap_size - 1) 

165 y[y_boundary_inds] = torch.tensor(heatmap_size - 1) 

166 

167 valid_loc = (x >= 0) & (y >= 0) & (x < heatmap_size) & (y < heatmap_size) 

168 vis = keypoints[..., 2] > 0 

169 valid = (valid_loc & vis).long() 

170 

171 lin_ind = y * heatmap_size + x 

172 heatmaps = lin_ind * valid 

173 

174 return heatmaps, valid

177def _onnx_heatmaps_to_keypoints(maps, maps_i, roi_map_width, roi_map_height, 

178 widths_i, heights_i, offset_x_i, offset_y_i): 

179 num_keypoints = torch.scalar_tensor(maps.size(1), dtype=torch.int64) 

180 

181 width_correction = widths_i / roi_map_width 

182 height_correction = heights_i / roi_map_height 

183 

184 roi_map = torch.nn.functional.interpolate( 

185 maps_i[None], size=(int(roi_map_height), int(roi_map_width)), mode='bicubic', align_corners=False)[0] 

186 

187 w = torch.scalar_tensor(roi_map.size(2), dtype=torch.int64) 

188 pos = roi_map.reshape(num_keypoints, -1).argmax(dim=1) 

189 

190 x_int = (pos % w) 

191 y_int = ((pos - x_int) / w) 

192 

193 x = (torch.tensor(0.5, dtype=torch.float32) + x_int.to(dtype=torch.float32)) * \ 

194 width_correction.to(dtype=torch.float32) 

195 y = (torch.tensor(0.5, dtype=torch.float32) + y_int.to(dtype=torch.float32)) * \ 

196 height_correction.to(dtype=torch.float32) 

197 

198 xy_preds_i_0 = x + offset_x_i.to(dtype=torch.float32) 

199 xy_preds_i_1 = y + offset_y_i.to(dtype=torch.float32) 

200 xy_preds_i_2 = torch.ones((xy_preds_i_1.shape), dtype=torch.float32) 

201 xy_preds_i = torch.stack([xy_preds_i_0.to(dtype=torch.float32), 

202 xy_preds_i_1.to(dtype=torch.float32), 

203 xy_preds_i_2.to(dtype=torch.float32)], 0) 

204 

205 # TODO: simplify when indexing without rank will be supported by ONNX 

206 end_scores_i = roi_map.index_select(1, y_int.to(dtype=torch.int64)) \ 

207 .index_select(2, x_int.to(dtype=torch.int64))[:num_keypoints, 0, 0] 

208 return xy_preds_i, end_scores_i

240def heatmaps_to_keypoints(maps, rois): 

241 """Extract predicted keypoint locations from heatmaps. Output has shape 

242 (#rois, 4, #keypoints) with the 4 rows corresponding to (x, y, logit, prob) 

243 for each keypoint. 

244 """ 

245 # This function converts a discrete image coordinate in a HEATMAP_SIZE x 

246 # HEATMAP_SIZE image to a continuous keypoint coordinate. We maintain 

247 # consistency with keypoints_to_heatmap_labels by using the conversion from 

248 # Heckbert 1990: c = d + 0.5, where d is a discrete coordinate and c is a 

249 # continuous coordinate. 

250 offset_x = rois[:, 0] 

251 offset_y = rois[:, 1] 

252 

253 widths = rois[:, 2] - rois[:, 0] 

254 heights = rois[:, 3] - rois[:, 1] 

255 widths = widths.clamp(min=1) 

256 heights = heights.clamp(min=1) 

257 widths_ceil = widths.ceil() 

258 heights_ceil = heights.ceil() 

259 

260 num_keypoints = maps.shape[1] 

261 

262 if torchvision._is_tracing(): 

263 xy_preds, end_scores = _onnx_heatmaps_to_keypoints_loop(maps, rois, 

264 widths_ceil, heights_ceil, widths, heights, 

265 offset_x, offset_y, 

266 torch.scalar_tensor(num_keypoints, dtype=torch.int64)) 

267 return xy_preds.permute(0, 2, 1), end_scores 

268 

269 xy_preds = torch.zeros((len(rois), 3, num_keypoints), dtype=torch.float32, device=maps.device) 

270 end_scores = torch.zeros((len(rois), num_keypoints), dtype=torch.float32, device=maps.device) 

271 for i in range(len(rois)): 

272 roi_map_width = int(widths_ceil[i].item()) 

273 roi_map_height = int(heights_ceil[i].item()) 

274 width_correction = widths[i] / roi_map_width 

275 height_correction = heights[i] / roi_map_height 

276 roi_map = torch.nn.functional.interpolate( 

277 maps[i][None], size=(roi_map_height, roi_map_width), mode='bicubic', align_corners=False)[0] 

278 # roi_map_probs = scores_to_probs(roi_map.copy()) 

279 w = roi_map.shape[2] 

280 pos = roi_map.reshape(num_keypoints, -1).argmax(dim=1) 

281 

282 x_int = pos % w 

283 y_int = tensor_floordiv((pos - x_int), w) 

284 # assert (roi_map_probs[k, y_int, x_int] == 

285 # roi_map_probs[k, :, :].max()) 

286 x = (x_int.float() + 0.5) * width_correction 

287 y = (y_int.float() + 0.5) * height_correction 

288 xy_preds[i, 0, :] = x + offset_x[i] 

289 xy_preds[i, 1, :] = y + offset_y[i] 

290 xy_preds[i, 2, :] = 1 

291 end_scores[i, :] = roi_map[torch.arange(num_keypoints), y_int, x_int] 

292 

293 return xy_preds.permute(0, 2, 1), end_scores

297 # type: (Tensor, List[Tensor], List[Tensor], List[Tensor])

321 

322 keypoint_loss = F.cross_entropy(keypoint_logits[valid], keypoint_targets[valid])

348def _onnx_expand_boxes(boxes, scale): 

349 # type: (Tensor, float) 

350 w_half = (boxes[:, 2] - boxes[:, 0]) * .5 

351 h_half = (boxes[:, 3] - boxes[:, 1]) * .5 

352 x_c = (boxes[:, 2] + boxes[:, 0]) * .5 

353 y_c = (boxes[:, 3] + boxes[:, 1]) * .5 

354 

355 w_half = w_half.to(dtype=torch.float32) * scale 

356 h_half = h_half.to(dtype=torch.float32) * scale 

357 

358 boxes_exp0 = x_c - w_half 

359 boxes_exp1 = y_c - h_half 

360 boxes_exp2 = x_c + w_half 

361 boxes_exp3 = y_c + h_half 

362 boxes_exp = torch.stack((boxes_exp0, boxes_exp1, boxes_exp2, boxes_exp3), 1) 

363 return boxes_exp

433def _onnx_paste_mask_in_image(mask, box, im_h, im_w): 

434 one = torch.ones(1, dtype=torch.int64) 

435 zero = torch.zeros(1, dtype=torch.int64) 

436 

437 w = (box[2] - box[0] + one) 

438 h = (box[3] - box[1] + one) 

439 w = torch.max(torch.cat((w, one))) 

440 h = torch.max(torch.cat((h, one))) 

441 

442 # Set shape to [batchxCxHxW] 

443 mask = mask.expand((1, 1, mask.size(0), mask.size(1))) 

444 

445 # Resize mask 

446 mask = torch.nn.functional.interpolate(mask, size=(int(h), int(w)), mode='bilinear', align_corners=False) 

447 mask = mask[0][0] 

448 

449 x_0 = torch.max(torch.cat((box[0].unsqueeze(0), zero))) 

450 x_1 = torch.min(torch.cat((box[2].unsqueeze(0) + one, im_w.unsqueeze(0)))) 

451 y_0 = torch.max(torch.cat((box[1].unsqueeze(0), zero))) 

452 y_1 = torch.min(torch.cat((box[3].unsqueeze(0) + one, im_h.unsqueeze(0))))

459 # pad y 

460 zeros_y0 = torch.zeros(y_0, unpaded_im_mask.size(1)) 

461 zeros_y1 = torch.zeros(im_h - y_1, unpaded_im_mask.size(1))

583 def assign_targets_to_proposals(self, proposals, gt_boxes, gt_labels): 

584 # type: (List[Tensor], List[Tensor], List[Tensor]) 

585 matched_idxs = [] 

586 labels = [] 

587 for proposals_in_image, gt_boxes_in_image, gt_labels_in_image in zip(proposals, gt_boxes, gt_labels): 

588 # set to self.box_similarity when https://github.com/pytorch/pytorch/issues/27495 lands 

589 match_quality_matrix = box_ops.box_iou(gt_boxes_in_image, proposals_in_image) 

590 matched_idxs_in_image = self.proposal_matcher(match_quality_matrix) 

591 

592 clamped_matched_idxs_in_image = matched_idxs_in_image.clamp(min=0) 

593 

594 labels_in_image = gt_labels_in_image[clamped_matched_idxs_in_image] 

595 labels_in_image = labels_in_image.to(dtype=torch.int64) 

596 

597 # Label background (below the low threshold) 

598 bg_inds = matched_idxs_in_image == self.proposal_matcher.BELOW_LOW_THRESHOLD 

599 labels_in_image[bg_inds] = torch.tensor(0) 

600 

601 # Label ignore proposals (between low and high thresholds) 

602 ignore_inds = matched_idxs_in_image == self.proposal_matcher.BETWEEN_THRESHOLDS 

603 labels_in_image[ignore_inds] = torch.tensor(-1) # -1 is ignored by sampler 

604 

605 matched_idxs.append(clamped_matched_idxs_in_image) 

606 labels.append(labels_in_image) 

607 return matched_idxs, labels

609 def subsample(self, labels): 

610 # type: (List[Tensor]) 

611 sampled_pos_inds, sampled_neg_inds = self.fg_bg_sampler(labels) 

612 sampled_inds = [] 

613 for img_idx, (pos_inds_img, neg_inds_img) in enumerate( 

614 zip(sampled_pos_inds, sampled_neg_inds) 

615 ): 

616 img_sampled_inds = torch.nonzero(pos_inds_img | neg_inds_img).squeeze(1) 

617 sampled_inds.append(img_sampled_inds) 

618 return sampled_inds

644 def select_training_samples(self, proposals, targets): 

645 # type: (List[Tensor], Optional[List[Dict[str, Tensor]]]) 

646 self.check_targets(targets) 

647 assert targets is not None 

648 dtype = proposals[0].dtype 

649 gt_boxes = [t["boxes"].to(dtype) for t in targets] 

650 gt_labels = [t["labels"] for t in targets] 

651 

652 # append ground-truth bboxes to propos 

653 proposals = self.add_gt_proposals(proposals, gt_boxes) 

654 

655 # get matching gt indices for each proposal 

656 matched_idxs, labels = self.assign_targets_to_proposals(proposals, gt_boxes, gt_labels) 

657 # sample a fixed proportion of positive-negative proposals 

658 sampled_inds = self.subsample(labels) 

659 matched_gt_boxes = [] 

660 num_images = len(proposals) 

661 for img_id in range(num_images): 

662 img_sampled_inds = sampled_inds[img_id] 

663 proposals[img_id] = proposals[img_id][img_sampled_inds] 

664 labels[img_id] = labels[img_id][img_sampled_inds] 

665 matched_idxs[img_id] = matched_idxs[img_id][img_sampled_inds] 

666 matched_gt_boxes.append(gt_boxes[img_id][matched_idxs[img_id]]) 

667 

668 regression_targets = self.box_coder.encode(matched_gt_boxes, proposals) 

669 return proposals, matched_idxs, labels, regression_targets

671 def postprocess_detections(self, class_logits, box_regression, proposals, image_shapes): 

672 # type: (Tensor, Tensor, List[Tensor], List[Tuple[int, int]]) 

673 device = class_logits.device 

674 num_classes = class_logits.shape[-1] 

675 

676 boxes_per_image = [len(boxes_in_image) for boxes_in_image in proposals] 

677 pred_boxes = self.box_coder.decode(box_regression, proposals) 

678 

679 pred_scores = F.softmax(class_logits, -1) 

680 

681 # split boxes and scores per image 

682 if len(boxes_per_image) == 1: 

683 # TODO : remove this when ONNX support dynamic split sizes 

684 # and just assign to pred_boxes instead of pred_boxes_list 

685 pred_boxes_list = [pred_boxes] 

686 pred_scores_list = [pred_scores] 

687 else: 

688 pred_boxes_list = pred_boxes.split(boxes_per_image, 0) 

689 pred_scores_list = pred_scores.split(boxes_per_image, 0) 

690 

691 all_boxes = [] 

692 all_scores = [] 

693 all_labels = [] 

694 for boxes, scores, image_shape in zip(pred_boxes_list, pred_scores_list, image_shapes): 

695 boxes = box_ops.clip_boxes_to_image(boxes, image_shape) 

696 

697 # create labels for each prediction 

698 labels = torch.arange(num_classes, device=device) 

699 labels = labels.view(1, -1).expand_as(scores) 

700 

701 # remove predictions with the background label 

702 boxes = boxes[:, 1:] 

703 scores = scores[:, 1:] 

704 labels = labels[:, 1:] 

705 

706 # batch everything, by making every class prediction be a separate instance 

707 boxes = boxes.reshape(-1, 4) 

708 scores = scores.reshape(-1) 

709 labels = labels.reshape(-1) 

710 

711 # remove low scoring boxes 

712 inds = torch.nonzero(scores > self.score_thresh).squeeze(1) 

713 boxes, scores, labels = boxes[inds], scores[inds], labels[inds] 

714 

715 # remove empty boxes 

716 keep = box_ops.remove_small_boxes(boxes, min_size=1e-2) 

717 boxes, scores, labels = boxes[keep], scores[keep], labels[keep] 

718 

719 # non-maximum suppression, independently done per class 

720 keep = box_ops.batched_nms(boxes, scores, labels, self.nms_thresh) 

721 # keep only topk scoring predictions 

722 keep = keep[:self.detections_per_img] 

723 boxes, scores, labels = boxes[keep], scores[keep], labels[keep] 

724 

725 all_boxes.append(boxes) 

726 all_scores.append(scores) 

727 all_labels.append(labels) 

728 

729 return all_boxes, all_scores, all_labels
