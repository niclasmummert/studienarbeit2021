607class RandomResizedCrop(object): 

608 """Crop the given PIL Image to random size and aspect ratio. 

609 

610 A crop of random size (default: of 0.08 to 1.0) of the original size and a random 

611 aspect ratio (default: of 3/4 to 4/3) of the original aspect ratio is made. This crop 

612 is finally resized to given size. 

613 This is popularly used to train the Inception networks. 

614 

615 Args: 

616 size: expected output size of each edge 

617 scale: range of size of the origin size cropped 

618 ratio: range of aspect ratio of the origin aspect ratio cropped 

619 interpolation: Default: PIL.Image.BILINEAR 

620 """ 

621 

622 def __init__(self, size, scale=(0.08, 1.0), ratio=(3. / 4., 4. / 3.), interpolation=Image.BILINEAR): 

623 if isinstance(size, tuple): 

624 self.size = size 

625 else: 

626 self.size = (size, size) 

627 if (scale[0] > scale[1]) or (ratio[0] > ratio[1]): 

628 warnings.warn("range should be of kind (min, max)") 

629 

630 self.interpolation = interpolation 

631 self.scale = scale 

632 self.ratio = ratio

634 @staticmethod 

635 def get_params(img, scale, ratio): 

636 """Get parameters for ``crop`` for a random sized crop. 

637 

638 Args: 

639 img (PIL Image): Image to be cropped. 

640 scale (tuple): range of size of the origin size cropped 

641 ratio (tuple): range of aspect ratio of the origin aspect ratio cropped 

642 

643 Returns: 

644 tuple: params (i, j, h, w) to be passed to ``crop`` for a random 

645 sized crop. 

646 """ 

647 width, height = _get_image_size(img) 

648 area = height * width 

649 

650 for attempt in range(10): 

651 target_area = random.uniform(*scale) * area 

652 log_ratio = (math.log(ratio[0]), math.log(ratio[1])) 

653 aspect_ratio = math.exp(random.uniform(*log_ratio)) 

654 

655 w = int(round(math.sqrt(target_area * aspect_ratio))) 

656 h = int(round(math.sqrt(target_area / aspect_ratio))) 

657 

658 if 0 < w <= width and 0 < h <= height: 

659 i = random.randint(0, height - h) 

660 j = random.randint(0, width - w) 

661 return i, j, h, w

664 in_ratio = float(width) / float(height) 

665 if (in_ratio < min(ratio)): 

666 w = width 

667 h = int(round(w / min(ratio))) 

668 elif (in_ratio > max(ratio)): 

669 h = height 

670 w = int(round(h * max(ratio))) 

671 else: # whole image 

672 w = width 

673 h = height 

674 i = (height - h) // 2 

675 j = (width - w) // 2 

676 return i, j, h, w

685 """ 

686 i, j, h, w = self.get_params(img, self.scale, self.ratio) 

687 return F.resized_crop(img, i, j, h, w, self.size, self.interpolation)

790class LinearTransformation(object): 

791 """Transform a tensor image with a square transformation matrix and a mean_vector computed 

792 offline. 

793 Given transformation_matrix and mean_vector, will flatten the torch.*Tensor and 

794 subtract mean_vector from it which is then followed by computing the dot 

795 product with the transformation matrix and then reshaping the tensor to its 

796 original shape. 

797 

798 Applications: 

799 whitening transformation: Suppose X is a column vector zero-centered data. 

800 Then compute the data covariance matrix [D x D] with torch.mm(X.t(), X), 

801 perform SVD on this matrix and pass it as transformation_matrix. 

802 

803 Args: 

804 transformation_matrix (Tensor): tensor [D x D], D = C x H x W 

805 mean_vector (Tensor): tensor [D], D = C x H x W 

806 """ 

807 

808 def __init__(self, transformation_matrix, mean_vector): 

809 if transformation_matrix.size(0) != transformation_matrix.size(1): 

810 raise ValueError("transformation_matrix should be square. Got " + 

811 "[{} x {}] rectangular matrix.".format(*transformation_matrix.size())) 

812 

813 if mean_vector.size(0) != transformation_matrix.size(0): 

814 raise ValueError("mean_vector should have the same length {}".format(mean_vector.size(0)) + 

815 " as any one of the dimensions of the transformation_matrix [{} x {}]" 

816 .format(transformation_matrix.size())) 

817 

818 self.transformation_matrix = transformation_matrix 

819 self.mean_vector = mean_vector 

820 

821 def __call__(self, tensor): 

822 """ 

823 Args: 

824 tensor (Tensor): Tensor image of size (C, H, W) to be whitened. 

825 

826 Returns: 

827 Tensor: Transformed image. 

828 """ 

829 if tensor.size(0) * tensor.size(1) * tensor.size(2) != self.transformation_matrix.size(0): 

830 raise ValueError("tensor and transformation matrix have incompatible shape." + 

831 "[{} x {} x {}] != ".format(*tensor.size()) + 

832 "{}".format(self.transformation_matrix.size(0))) 

833 flat_tensor = tensor.view(1, -1) - self.mean_vector 

834 transformed_tensor = torch.mm(flat_tensor, self.transformation_matrix) 

835 tensor = transformed_tensor.view(tensor.size()) 

836 return tensor

839 format_string = self.__class__.__name__ + '(transformation_matrix=' 

840 format_string += (str(self.transformation_matrix.tolist()) + ')')

845class ColorJitter(object): 

846 """Randomly change the brightness, contrast and saturation of an image. 

847 

848 Args: 

849 brightness (float or tuple of float (min, max)): How much to jitter brightness. 

850 brightness_factor is chosen uniformly from [max(0, 1 - brightness), 1 + brightness] 

851 or the given [min, max]. Should be non negative numbers. 

852 contrast (float or tuple of float (min, max)): How much to jitter contrast. 

853 contrast_factor is chosen uniformly from [max(0, 1 - contrast), 1 + contrast] 

854 or the given [min, max]. Should be non negative numbers.

864 self.contrast = self._check_input(contrast, 'contrast')

871 if value < 0: 

872 raise ValueError("If {} is a single number, it must be non negative.".format(name)) 

873 value = [center - value, center + value] 

874 if clip_first_on_zero: 

875 value[0] = max(value[0], 0) 

876 elif isinstance(value, (tuple, list)) and len(value) == 2: 

877 if not bound[0] <= value[0] <= value[1] <= bound[1]: 

878 raise ValueError("{} values should be between {}".format(name, bound)) 

879 else: 

880 raise TypeError("{} should be a single number or a list/tuple with lenght 2.".format(name)) 

881 

882 # if value is 0 or (1., 1.) for brightness/contrast/saturation 

883 # or (0., 0.) for hue, do nothing 

884 if value[0] == value[1] == center: 

885 value = None 

886 return value
