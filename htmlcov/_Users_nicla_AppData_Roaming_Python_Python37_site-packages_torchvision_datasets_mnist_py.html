416def open_maybe_compressed_file(path): 

417 """Return a file object that possibly decompresses 'path' on the fly. 

418 Decompression occurs when argument `path` is a string and ends with '.gz' or '.xz'. 

419 """ 

420 if not isinstance(path, torch._six.string_classes): 

421 return path 

422 if path.endswith('.gz'): 

423 import gzip 

424 return gzip.open(path, 'rb') 

425 if path.endswith('.xz'): 

426 import lzma 

427 return lzma.open(path, 'rb') 

428 return open(path, 'rb')

431def read_sn3_pascalvincent_tensor(path, strict=True): 

432 """Read a SN3 file in "Pascal Vincent" format (Lush file 'libidx/idx-io.lsh'). 

433 Argument may be a filename, compressed filename, or file object. 

434 """ 

435 # typemap 

436 if not hasattr(read_sn3_pascalvincent_tensor, 'typemap'): 

437 read_sn3_pascalvincent_tensor.typemap = { 

438 8: (torch.uint8, np.uint8, np.uint8), 

439 9: (torch.int8, np.int8, np.int8), 

440 11: (torch.int16, np.dtype('>i2'), 'i2'), 

441 12: (torch.int32, np.dtype('>i4'), 'i4'), 

442 13: (torch.float32, np.dtype('>f4'), 'f4'), 

443 14: (torch.float64, np.dtype('>f8'), 'f8')} 

444 # read 

445 with open_maybe_compressed_file(path) as f: 

446 data = f.read() 

447 # parse 

448 magic = get_int(data[0:4]) 

449 nd = magic % 256 

450 ty = magic // 256 

451 assert nd >= 1 and nd <= 3 

452 assert ty >= 8 and ty <= 14 

453 m = read_sn3_pascalvincent_tensor.typemap[ty] 

454 s = [get_int(data[4 * (i + 1): 4 * (i + 2)]) for i in range(nd)] 

455 parsed = np.frombuffer(data, dtype=m[1], offset=(4 * (nd + 1))) 

456 assert parsed.shape[0] == np.prod(s) or not strict 

457 return torch.from_numpy(parsed.astype(m[2], copy=False)).view(*s)
