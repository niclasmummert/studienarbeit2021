64 path where the video will be saved

95def _read_from_stream(container, start_offset, end_offset, pts_unit, stream, stream_name): 

96 global _CALLED_TIMES, _GC_COLLECTION_INTERVAL 

97 _CALLED_TIMES += 1 

98 if _CALLED_TIMES % _GC_COLLECTION_INTERVAL == _GC_COLLECTION_INTERVAL - 1: 

99 gc.collect() 

100 

101 if pts_unit == 'sec': 

102 start_offset = int(math.floor(start_offset * (1 / stream.time_base))) 

103 if end_offset != float("inf"): 

104 end_offset = int(math.ceil(end_offset * (1 / stream.time_base))) 

105 else: 

106 warnings.warn("The pts_unit 'pts' gives wrong results and will be removed in a " + 

107 "follow-up version. Please use pts_unit 'sec'.") 

108 

109 frames = {} 

110 should_buffer = False 

111 max_buffer_size = 5 

112 if stream.type == "video": 

113 # DivX-style packed B-frames can have out-of-order pts (2 frames in a single pkt) 

114 # so need to buffer some extra frames to sort everything 

115 # properly 

116 extradata = stream.codec_context.extradata 

117 # overly complicated way of finding if `divx_packed` is set, following 

118 # https://github.com/FFmpeg/FFmpeg/commit/d5a21172283572af587b3d939eba0091484d3263 

119 if extradata and b"DivX" in extradata: 

120 # can't use regex directly because of some weird characters sometimes... 

121 pos = extradata.find(b"DivX") 

122 d = extradata[pos:] 

123 o = re.search(br"DivX(\d+)Build(\d+)(\w)", d) 

124 if o is None: 

125 o = re.search(br"DivX(\d+)b(\d+)(\w)", d) 

126 if o is not None: 

127 should_buffer = o.group(3) == b"p" 

128 seek_offset = start_offset 

129 # some files don't seek to the right location, so better be safe here 

130 seek_offset = max(seek_offset - 1, 0) 

131 if should_buffer: 

132 # FIXME this is kind of a hack, but we will jump to the previous keyframe 

133 # so this will be safe 

134 seek_offset = max(seek_offset - max_buffer_size, 0) 

135 try: 

136 # TODO check if stream needs to always be the video stream here or not 

137 container.seek(seek_offset, any_frame=False, backward=True, stream=stream) 

138 except av.AVError: 

139 # TODO add some warnings in this case 

140 # print("Corrupted file?", container.name) 

141 return [] 

142 buffer_count = 0 

143 try: 

144 for idx, frame in enumerate(container.decode(**stream_name)): 

145 frames[frame.pts] = frame 

146 if frame.pts >= end_offset: 

147 if should_buffer and buffer_count < max_buffer_size: 

148 buffer_count += 1 

149 continue 

150 break 

151 except av.AVError: 

152 # TODO add a warning 

153 pass 

154 # ensure that the results are sorted wrt the pts 

155 result = [frames[i] for i in sorted(frames) if start_offset <= frames[i].pts <= end_offset] 

156 if len(frames) > 0 and start_offset > 0 and start_offset not in frames: 

157 # if there is no frame that exactly matches the pts of start_offset 

158 # add the last frame smaller than start_offset, to guarantee that 

159 # we will have all the necessary data. This is most useful for audio 

160 preceding_frames = [i for i in frames if i < start_offset] 

161 if len(preceding_frames) > 0: 

162 first_frame_pts = max(preceding_frames) 

163 result.insert(0, frames[first_frame_pts]) 

164 return result

167def _align_audio_frames(aframes, audio_frames, ref_start, ref_end): 

168 start, end = audio_frames[0].pts, audio_frames[-1].pts 

169 total_aframes = aframes.shape[1] 

170 step_per_aframe = (end - start + 1) / total_aframes 

171 s_idx = 0 

172 e_idx = total_aframes 

173 if start < ref_start: 

174 s_idx = int((ref_start - start) / step_per_aframe) 

175 if end > ref_end: 

176 e_idx = int((ref_end - end) / step_per_aframe) 

177 return aframes[:, s_idx:e_idx]

180def read_video(filename, start_pts=0, end_pts=None, pts_unit='pts'): 

181 """ 

182 Reads a video from a file, returning both the video frames as well as 

183 the audio frames 

184 

185 Parameters 

186 ---------- 

187 filename : str 

188 path to the video file 

189 start_pts : int if pts_unit = 'pts', optional 

190 float / Fraction if pts_unit = 'sec', optional 

191 the start presentation time of the video 

192 end_pts : int if pts_unit = 'pts', optional 

193 float / Fraction if pts_unit = 'sec', optional 

194 the end presentation time 

195 pts_unit : str, optional 

196 unit in which start_pts and end_pts values will be interpreted, either 'pts' or 'sec'. Defaults to 'pts'. 

197 

198 Returns 

199 ------- 

200 vframes : Tensor[T, H, W, C] 

201 the `T` video frames 

202 aframes : Tensor[K, L] 

203 the audio frames, where `K` is the number of channels and `L` is the 

204 number of points 

205 info : Dict 

206 metadata for the video and audio. Can contain the fields video_fps (float) 

207 and audio_fps (int) 

208 """ 

209 

210 from torchvision import get_video_backend 

211 if get_video_backend() != "pyav": 

212 return _video_opt._read_video(filename, start_pts, end_pts, pts_unit)

214 _check_av_available() 

215 

216 if end_pts is None: 

217 end_pts = float("inf") 

218 

219 if end_pts < start_pts: 

220 raise ValueError("end_pts should be larger than start_pts, got " 

221 "start_pts={} and end_pts={}".format(start_pts, end_pts)) 

222 

223 info = {} 

224 video_frames = [] 

225 audio_frames = [] 

226 

227 try: 

228 container = av.open(filename, metadata_errors='ignore') 

229 except av.AVError:

232 else:

235 container.streams.video[0], {'video': 0})

237 # guard against potentially corrupted files 

238 if video_fps is not None: 

239 info["video_fps"] = float(video_fps) 

240 

241 if container.streams.audio: 

242 audio_frames = _read_from_stream(container, start_pts, end_pts, pts_unit,

246 container.close()

249 aframes = [frame.to_ndarray() for frame in audio_frames]

251 if vframes:

254 vframes = torch.empty((0, 1, 1, 3), dtype=torch.uint8)

256 if aframes: 

257 aframes = np.concatenate(aframes, 1) 

258 aframes = torch.as_tensor(aframes) 

259 aframes = _align_audio_frames(aframes, audio_frames, start_pts, end_pts)
