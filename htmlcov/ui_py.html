41def convert_to_bytes(file_or_bytes, resize=None): 

42 ''' 

43 Will convert into bytes and optionally resize an image that is a file or a base64 bytes object. 

44 Turns into PNG format in the process so that can be displayed by tkinter 

45 :param file_or_bytes: either a string filename or a bytes base64 image object 

46 :type file_or_bytes: (Union[str, bytes]) 

47 :param resize: optional new size 

48 :type resize: (Tuple[int, int] or None) 

49 :return: (bytes) a byte-string object 

50 :rtype: (bytes) 

51 ''' 

52 if isinstance(file_or_bytes, str): 

53 img = PIL.Image.open(file_or_bytes) 

54 else: 

55 try: 

56 img = PIL.Image.open(io.BytesIO(base64.b64decode(file_or_bytes))) 

57 except Exception as e: 

58 dataBytesIO = io.BytesIO(file_or_bytes) 

59 img = PIL.Image.open(dataBytesIO) 

60 

61 cur_width, cur_height = img.size 

62 if resize: 

63 new_width, new_height = resize 

64 scale = min(new_height/cur_height, new_width/cur_width) 

65 img = img.resize((int(cur_width*scale), int(cur_height*scale)), PIL.Image.ANTIALIAS) 

66 with io.BytesIO() as bio: 

67 img.save(bio, format="PNG") 

68 del img 

69 return bio.getvalue()
