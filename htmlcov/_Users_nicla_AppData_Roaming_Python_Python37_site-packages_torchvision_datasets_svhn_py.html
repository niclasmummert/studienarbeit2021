10class SVHN(VisionDataset): 

11 """`SVHN <http://ufldl.stanford.edu/housenumbers/>`_ Dataset. 

12 Note: The SVHN dataset assigns the label `10` to the digit `0`. However, in this Dataset, 

13 we assign the label `0` to the digit `0` to be compatible with PyTorch loss functions which 

14 expect the class labels to be in the range `[0, C-1]` 

15 

16 Args: 

17 root (string): Root directory of dataset where directory 

18 ``SVHN`` exists. 

19 split (string): One of {'train', 'test', 'extra'}. 

20 Accordingly dataset is selected. 'extra' is Extra training set. 

21 transform (callable, optional): A function/transform that takes in an PIL image 

22 and returns a transformed version. E.g, ``transforms.RandomCrop`` 

23 target_transform (callable, optional): A function/transform that takes in the 

24 target and transforms it. 

25 download (bool, optional): If true, downloads the dataset from the internet and 

26 puts it in root directory. If dataset is already downloaded, it is not 

27 downloaded again. 

28 

29 """ 

30

59 # reading(loading) mat file as array 

60 loaded_mat = sio.loadmat(os.path.join(self.root, self.filename)) 

61 

62 self.data = loaded_mat['X'] 

63 # loading from the .mat file gives an np array of type np.uint8 

64 # converting to np.int64, so that we have a LongTensor after 

65 # the conversion from the numpy array 

66 # the squeeze is needed to obtain a 1D tensor 

67 self.labels = loaded_mat['y'].astype(np.int64).squeeze() 

68 

69 # the svhn dataset assigns the class label "10" to the digit 0 

70 # this makes it inconsistent with several loss functions 

71 # which expect the class labels to be in the range [0, C-1] 

72 np.place(self.labels, self.labels == 10, 0) 

73 self.data = np.transpose(self.data, (3, 2, 0, 1))
